country_event = {
	id = GOE_RAJ_partition.11
	title = GOE_RAJ_partition.7.t #Intended
	desc = GOE_RAJ_partition.11.desc
	picture = GFX_report_event_generic_sign_treaty2
	is_triggered_only = yes
	option = { #Nehru
		name = GOE_RAJ_partition.7.a #Intended

		ai_chance = {
			base = 10
		}

		if = {
			limit = {
				NOT = {
					has_country_leader = {
						ruling_only = yes
						character = RAJ_GOE_jawaharlal_nehru 
					}
				}
			}
			add_country_leader_role = {
				character =  RAJ_GOE_jawaharlal_nehru 
				promote_leader = yes
				country_leader={
					ideology=liberalism
					expire="1965.1.1.1"
					traits = { socialist_modernizer top_down_heavy_policies }
				}
			}
			set_politics = {
				ruling_party = democratic
				election_frequency = 48
				elections_allowed = yes
			}
		}
		else = {
			add_country_leader_trait = socialist_modernizer
			add_country_leader_trait = top_down_heavy_policies
		}
	}

	#Gandhi
	option = {
		name =  GOE_RAJ_partition.7.b #Intended

		ai_chance = {
			base = 0
			modifier = {
				is_historical_focus_on = no
				add = 10
			}
		}

		if = {
			limit = {
				NOT = { has_government = democratic }
			}
			set_politics = {
				ruling_party = democratic
				election_frequency = 48
				elections_allowed = yes
			}
		}
		custom_effect_tooltip = generic_skip_one_line_tt

		#Tooltip will be just bad, so simplifying
		effect_tooltip = {
			add_country_leader_role = {
				character =  RAJ_mahatma_mohandas_gandhi
				promote_leader = yes
				country_leader={
					ideology=liberalism
					expire="1965.1.1.1"
					traits = { spiritual_leader skeptical_towards_industrialization }
				}
			}
		}
		hidden_effect = {
			if = { #If no Country Leader role for whatever reason, add it.
				limit = {
					RAJ_mahatma_mohandas_gandhi = {
						NOT = { has_ideology = liberalism }
					}
				}
				add_country_leader_role = {
					character =  RAJ_mahatma_mohandas_gandhi
					promote_leader = yes
					country_leader={
						ideology=liberalism
						expire="1965.1.1.1"
						traits = { spiritual_leader skeptical_towards_industrialization }
					}
				}
			}
			else = {
				# Promote Gandhi if necessary
				if = {
					limit = {
						NOT = {
							has_country_leader = {
								ruling_only = no
								character = RAJ_mahatma_mohandas_gandhi
							}
						}
					}
					promote_character = RAJ_mahatma_mohandas_gandhi 
				}

				# Add Traits
				if = {
					limit = {
						NOT = { has_country_leader_with_trait = spiritual_leader }
					}
					add_country_leader_trait = spiritual_leader
				}
				if = {
					limit = {
						NOT = { has_country_leader_with_trait = skeptical_towards_industrialization }
					}
					add_country_leader_trait = skeptical_towards_industrialization
				}
			}
		}
	}
}

	id = GOE_RAJ_loyalist.02

	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			random_core_state = {
				limit = {
					is_fully_controlled_by = PREV
					free_building_slots = {
						building = industrial_complex
						size > 1
						include_locked = yes
					}
				}
				set_variable = { ROOT.RAJ_crown_commodities_civ_built_flag = THIS }
			}
		}
	}
	
	option = { #Ok cool tack
		name = GOE_RAJ_loyalist.02.a

		if = {
			limit = {
				tag = ENG
			}
			var:RAJ_crown_commodities_civ_built_flag = {
				add_extra_state_shared_building_slots = 2
				add_building_construction = {
					type = industrial_complex
					level = 2
					instant_build = yes
				}
			}
		}
		else = {
			var:RAJ_crown_commodities_civ_built_flag = {
				add_extra_state_shared_building_slots = 1
				add_building_construction = {
					type = industrial_complex
					level = 1
					instant_build = yes
				}
			}
		}
		clear_variable = RAJ_crown_commodities_civ_built_flag
	}
}
