				limit = { british_coastline_not_south = yes }

					if = {
						limit = { NOT = { OR = { state = 582 state = 323 } } }

						limit = { state = 582 POL = { owns_state = 582 } }

			if = { 
				limit = { FROM = { is_core_of = SPR } }
				teleport_armies = { limit = { tag = SPR } to_state = 41 } 
			}

			# Specific state teleport for Axis if date < 1940.1.1
			if = { 
				limit = { OR = { state = 313 state = 21 } date < 1940.1.1 }
