DIPLOMACY_LEND_LEASE_ENABLE_TRIGGER = {
	NOT = { date < 1939.9.1 } # dont remove bc of SCW!

	custom_trigger_tooltip = {
		tooltip = DIPLOMACY_LEND_LEASE_MASTER_tt
	}
	
	hidden_trigger = {
		OR = {
			# SCENARIO 1: The Allied Loop
			# Both sender and receiver must be ENG or in a faction with ENG
			AND = {
				ROOT = {
					OR = {
						original_tag = ENG
						is_in_faction_with = ENG
					}
				}
				FROM = {
					OR = {
						original_tag = ENG
						is_in_faction_with = ENG
					}
				}
			}

			# SCENARIO 2: The Axis Loop
			# Both sender and receiver must be GER or in a faction with GER
			AND = {
				ROOT = {
					OR = {
						original_tag = GER
						is_in_faction_with = GER
					}
				}
				FROM = {
					OR = {
						original_tag = GER
						is_in_faction_with = GER
					}
				}
			}

			# SCENARIO 3: The Co-Prosperity Loop
			# Both sender and receiver must be JAP or in a faction with JAP
			AND = {
				ROOT = {
					OR = {
						original_tag = JAP
						is_in_faction_with = JAP
					}
				}
				FROM = {
					OR = {
						original_tag = JAP
						is_in_faction_with = JAP
					}
				}
			}

			# SCENARIO 4: The Comintern Loop
			# Both sender and receiver must be SOV or in a faction with SOV
			AND = {
				ROOT = {
					OR = {
						original_tag = SOV
						is_in_faction_with = SOV
					}
				}
				FROM = {
					OR = {
						original_tag = SOV
						is_in_faction_with = SOV
					}
				}
			}

			# SCENARIO 5: The Rest of the World
			# If neither country belongs to the ENG, GER, JAP, or SOV spheres, 
			# they are allowed to lend-lease normally (e.g., USA to Brazil).
			AND = {
				ROOT = {
					NOT = { original_tag = ENG }
					NOT = { is_in_faction_with = ENG }
					NOT = { original_tag = GER }
					NOT = { is_in_faction_with = GER }
					NOT = { original_tag = JAP }
					NOT = { is_in_faction_with = JAP }
					NOT = { original_tag = SOV }
					NOT = { is_in_faction_with = SOV }
				}
				FROM = {
					NOT = { original_tag = ENG }
					NOT = { is_in_faction_with = ENG }
					NOT = { original_tag = GER }
					NOT = { is_in_faction_with = GER }
					NOT = { original_tag = JAP }
					NOT = { is_in_faction_with = JAP }
					NOT = { original_tag = SOV }
					NOT = { is_in_faction_with = SOV }
				}
			}
		}
		
		if = { 
			limit = { ROOT = { tag = FIN } } 
			date > 1940.1.1
		}

		if = { 
			limit = { ROOT = { tag = SOV } } 
			date > 1940.1.1
		}
		
		if = { 
			limit = { ROOT = { tag = MON } } 
			date > 1940.1.1
		}

		if = { 
			limit = { ROOT = { tag = USA } } 
			has_war = yes 
		}
	}
}
