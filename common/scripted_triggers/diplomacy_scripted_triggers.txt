	NOT = { date < 1939.1.1 } # dont remove bc of SCW!

	if = { 
		limit = { FROM = { tag = FRA } }  
		FRA = { has_country_flag = france_is_kill }  
	} 

	if = { 
		limit = { FROM = { tag = NOR } }  
		NOR = { has_country_flag = france_is_kill }  
	} 

	if = { 
		limit = { ROOT = { tag = USA } } 
		has_war = yes 
	}

	if = { 
		limit = { FROM = { the_axis = yes } } 
		NOT = { ROOT = { the_geacps = yes } } 
	}

	if = { 
		limit = { FROM = { normally_played_allies = yes } } 
		NOT = { ROOT = { the_geacps = yes } } 
	}

	if = { 
		limit = { ROOT = { the_axis = yes } } 
		NOT = { FROM = { the_geacps = yes } } 
	}

	NOT = { 
		ROOT = { 
			OR = { 
				tag = SOV  
				AND = { 
					original_tag = FIN 
					SOV = { controls_state = 195 } 
				} 
			} 
		} 
	}

	NOT = { 
		FROM = { 
			OR = { 
				tag = SOV  
				AND = { 
					original_tag = FIN 
					SOV = { controls_state = 195 } 
				} 
			} 
		} 
	}
	
	NOT = { 
		FROM = { 
			OR = { 
				tag = MON  
				AND = { 
					original_tag = FIN 
					SOV = { controls_state = 195 } 
				} 
			} 
		} 
	}
}
