        completion_reward = {
            # 1. Check which focus they did and set the base reward
            if = {
                limit = { has_completed_focus = SOV_the_plan_of_consolidation }
                set_variable = { horst_reward_amount = 12 }
            }
            else_if = {
                limit = { has_completed_focus = SOV_rearmament_efforts }
                set_variable = { horst_reward_amount = 18 }
            }

            # 2. Grab the current date and subtract the "Optimal Completion" baseline
            set_temp_variable = { temp_days = global.num_days }
            subtract_from_temp_variable = { temp_days = 706780 } 
            
            # Prevent negative math if they finish it perfectly on time
            if = {
                limit = { check_variable = { temp_days < 0 } }
                set_temp_variable = { temp_days = 0 }
            }

            # 3. Convert passed days to months and apply decay
            divide_temp_variable = { temp_days = 30 } 
            subtract_from_variable = { horst_reward_amount = temp_days }
            
            # 4. Clamp the variable so it never drops below 0
            if = {
                limit = { check_variable = { horst_reward_amount < 0 } }
                set_variable = { horst_reward_amount = 0 }
            }

            custom_effect_tooltip = horst_dynamic_factory_tt
            
            custom_effect_tooltip = generic_skip_one_line_tt
            
            custom_effect_tooltip = "Doing this focus will disable the decisions for the Second Five Year Plan."
            
            # 5. Route the factories based on prior focuses
            hidden_effect = {
                # ROUTE A: Plan of Consolidation = Civilian Factories
                if = {
                    limit = { has_completed_focus = SOV_the_plan_of_consolidation }
                    while_loop_effect = {
                        limit = { check_variable = { horst_reward_amount > 0 } }
                        random_owned_controlled_state = {
                            limit = { is_core_of = ROOT }
                            add_extra_state_shared_building_slots = 1
                            add_building_construction = {
                                type = industrial_complex
                                level = 1
                                instant_build = yes
                            }
                        }
                        subtract_from_variable = { horst_reward_amount = 1 }
                    }
                }
                
                # ROUTE B: Rearmament Efforts = Military Factories
                else_if = {
                    limit = { has_completed_focus = SOV_rearmament_efforts }
                    while_loop_effect = {
                        limit = { check_variable = { horst_reward_amount > 0 } }
                        random_owned_controlled_state = {
                            limit = { is_core_of = ROOT }
                            add_extra_state_shared_building_slots = 1
                            add_building_construction = {
                                type = arms_factory
                                level = 1
                                instant_build = yes
                            }
                        }
                        subtract_from_variable = { horst_reward_amount = 1 }
                    }
                }
                
                # 6. Clean up the variable
                clear_variable = horst_reward_amount
            }
        }
	}
